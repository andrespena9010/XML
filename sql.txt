select FIRST_NAME, LAST_NAME, SALARY from EMPLOYEES where SALARY > 12000;



select FIRST_NAME, LAST_NAME, SALARY from EMPLOYEES where SALARY not between 5000 and 12000;



select FIRST_NAME, LAST_NAME, MANAGER_ID, HIRE_DATE from EMPLOYEES where HIRE_DATE between '1995-02-20' and '1998-05-01' order by HIRE_DATE asc;




select FIRST_NAME, LAST_NAME, DEPARTMENT_ID from EMPLOYEES where DEPARTMENT_ID = 20 or DEPARTMENT_ID = 50 order by LAST_NAME asc;



select LAST_NAME, J.JOB_TITLE, SALARY from EMPLOYEES E , JOBS J where E.JOB_ID = J.JOB_ID AND (E.JOB_ID = 'SA_REP' OR E.JOB_ID = 'ST_CLERK') AND NOT E.SALARY = 2500;


SELECT MAX(SALARY) AS SALARIO_MAYOR, MIN(SALARY) AS SALARIO_MENOR, SUM(SALARY) AS SUMA_SALARIOS, ROUND(AVG(SALARY), 0) AS MEDIA_SALARIOS FROM EMPLOYEES;



SELECT JOB_ID, MAX(SALARY) AS SALARIO_MAYOR, MIN(SALARY) AS SALARIO_MENOR, SUM(SALARY) AS SUMA_SALARIOS, ROUND(AVG(SALARY), 0) AS MEDIA_SALARIOS FROM EMPLOYEES GROUP BY JOB_ID;



SELECT JOB_ID, COUNT(*) AS NUMERO_PERSONAS FROM EMPLOYEES GROUP BY JOB_ID HAVING AVG(SALARY) > 8000;



SELECT MAX(SALARY) AS SALARIO_MAXIMO, MIN(SALARY) AS SALARIO_MINIMO, (MAX(SALARY) - MIN(SALARY)) AS DIFERENCIA_SALARIOS FROM EMPLOYEES;


SELECT MAX(SALARY) AS SALARIO_MAXIMO, MIN(SALARY) AS SALARIO_MINIMO, (MAX(SALARY) - MIN(SALARY)) AS DIFERENCIA_SALARIOS FROM EMPLOYEES E, DEPARTMENTS D, LOCATIONS L, COUNTRIES C
WHERE E.DEPARTMENT_ID = D.DEPARTMENT_ID AND D.LOCATION_ID = L.LOCATION_ID AND L.COUNTRY_ID = C.COUNTRY_ID AND C.COUNTRY_ID = (SELECT COUNTRY_ID FROM COUNTRIES C WHERE C.COUNTRY_NAME = 'United States of America');
